$("#setupTOTPBtn").click(function () { if (!currentEmail) { showAlert("No user email found.", "danger"); return; } $.post("/setup_totp", { email: currentEmail }, function (res) { clearQRCode(); qrCodeInstance = new QRCode(document.getElementById("qrcode"), { text: res.qr_url, width: 200, height: 200, correctLevel: QRCode.CorrectLevel.H }); }).fail(function (xhr) { showAlert(xhr.responseJSON?.message || "Failed to setup TOTP", "danger"); }); }); $("#verify2FABtn").click(function () { const method = $('input[name="fa_method"]:checked').val(); let code = ''; if (method === 'totp') { code = $("#totpCode").val().trim(); } else if (method === 'email') { code = $("#emailCode").val().trim(); } if (!code) { showModalAlert("Please enter the verification code.", "warning"); return; } $.post("/verify_2fa", { email: currentEmail, code, method }, function (res) { if (res.message.includes("2FA verified successfully")) { showAlert("Login successful", "success"); $('#twoFactorModal').modal('hide'); setTimeout(() => { window.location.href = "/dashboard"; }, 1000); } else { showModalAlert(res.message, "danger"); } }).fail(function (xhr) { showModalAlert(xhr.responseJSON?.message || "Verification failed", "danger"); }); });
